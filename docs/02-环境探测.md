## Claude Code 路径解析

### Binary

```swift
func locateClaudeBinary() throws -> String {
    // 1. 在登录 shell 下跑 which claude（nvm/shim 解析）
    let result = try ShellRunner.run(
        command: "/bin/zsh -li -c 'which claude'",
        timeout: 3
    )
    if let path = result.stdout.trimmedOrNil, FileManager.isExecutable(path) {
        return path
    }

    // 2. 失败回退：扫描 nvm
    let nvmDir = FileManager.default.homeDirectoryForCurrentUser
        .appendingPathComponent(".nvm/versions/node")
    let candidates = try FileManager.default
        .contentsOfDirectory(at: nvmDir, includingPropertiesForKeys: nil)
        .filter { $0.lastPathComponent.hasPrefix("v") }
        .sorted(by: >)
        .map { $0.appendingPathComponent("bin/claude") }
        .filter { FileManager.isExecutable($0.path) }

    guard let fallback = candidates.first else {
        throw DetectionError.claudeNotFound
    }
    return fallback.path
}
```

**为什么用登录 shell？** Claude Code 通过 nvm shim 管理，`/usr/local/bin/claude` 不存在。`which claude` 只有在交互式 shell 加载 nvm 后才能解析。`-li`（login + interactive）保证加载 `~/.zshrc`。

**升级风险：** node 升级后路径里的 `v22.16.0` 会变。App 持久化路径时打 `nvm-managed` 标记，Poller 每次启动重跑 `which claude` 校正。

## 数据源识别（关键）

详见 [03-数据源策略.md](03-数据源策略.md)。Setup 阶段的核心检测：

```swift
func detectActiveSource() throws -> ActiveSource {
    let settings = try SettingsJsonReader.read(at: claudeConfigDir)

    // 1. 看 env 块有没有第三方 base URL
    if let baseURL = settings.env?.anthropicBaseURL,
       !isOfficialAnthropicURL(baseURL) {
        return .envVars(baseURL: baseURL)
    }

    // 2. 检测 cc switch
    if let ccSwitchDB = try? locateCCSwitchDB(),
       let db = try? CCSwitchDB.open(ccSwitchDB, readonly: true),
       let active = try? db.activeProvider(),
       active != nil {
        return .ccSwitch
    }

    // 3. 默认官方订阅
    return .official
}

func isOfficialAnthropicURL(_ url: String) -> Bool {
    let lowered = url.lowercased()
    return lowered.contains("anthropic.com") || lowered.contains("claude.ai")
}
```

### 三层数据源

| 层级 | 触发条件 | 提供什么 |
|---|---|---|
| **L1 官方订阅** | `settings.json` env 为空或指向 `anthropic.com` / `claude.ai` | 订阅状态推断（Pro/Max）、用量从 hook + transcript 自算 |
| **L2 第三方 env vars** | env 块含非官方 ANTHROPIC_BASE_URL | provider 名（从 URL 推断）、model 映射；cost 无数据 |
| **L3 cc switch** | 装了 cc switch 且当前 activeProvider 存在 | 完整 provider 列表、cost、健康状态、切换能力 |

## cc switch 路径解析（**最反直觉**）

### 1. App 路径

```swift
func locateCCSwitchApp() throws -> URL {
    let defaultPath = "/Applications/CC Switch.app"
    if FileManager.default.fileExists(atPath: defaultPath) {
        return URL(fileURLWithPath: defaultPath)
    }

    if let url = NSWorkspace.shared.urlForApplication(
        withBundleIdentifier: "com.ccswitch.desktop"
    ) {
        return url
    }

    throw DetectionError.ccSwitchAppNotFound
}
```

### 2. db 路径（**必须**读 override）

```swift
func locateCCSwitchDB() throws -> URL? {
    let appPathsURL = FileManager.default
        .homeDirectoryForCurrentUser
        .appendingPathComponent("Library/Application Support/com.ccswitch.desktop/app_paths.json")

    if let data = try? Data(contentsOf: appPathsURL),
       let appPaths = try? JSONDecoder().decode(AppPaths.self, from: data),
       let override = appPaths.appConfigDirOverride {
        let dbURL = URL(fileURLWithPath: override).appendingPathComponent("cc-switch.db")
        if FileManager.default.fileExists(atPath: dbURL.path) { return dbURL }
    }

    let defaultDB = FileManager.default
        .homeDirectoryForCurrentUser
        .appendingPathComponent(".cc-switch/cc-switch.db")
    return FileManager.default.fileExists(atPath: defaultDB.path) ? defaultDB : nil
}
```

### 3. 校验 db schema

```swift
func validateDB(_ url: URL) throws {
    let db = try CCSwitchDB.open(url, readonly: true)
    let tables = try db.query("SELECT name FROM sqlite_master WHERE type='table'")
    let required: Set = ["providers", "usage_daily_rollups", "proxy_config", "settings"]
    let actual = Set(tables.compactMap { $0["name"] as? String })
    guard required.isSubset(of: actual) else {
        throw DetectionError.schemaMismatch(missing: required.subtracting(actual))
    }
}
```

## 后续启动（非首次）

```swift
func applicationDidFinishLaunching(_ notification: Notification) {
    // 0. 机型检测
    guard isTouchBarAvailable() else {
        showUnsupportedModelAlert()
        NSApp.terminate(nil)
        return
    }

    let prefs = PreferenceStore.load()
    if !prefs.setupComplete {
        showSetupWizard()
        return
    }

    // 1. 重新探测数据源（settings.json 可能改了）
    appState.activeSource = (try? detectActiveSource()) ?? .official

    // 2. 校验 Claude binary
    if prefs.claudeBinaryIsNvmManaged {
        appState.claudeBinaryPath = try? ClaudeDetector.locateClaudeBinary()
    } else {
        appState.claudeBinaryPath = FileManager.isExecutable(prefs.claudeBinary) ? prefs.claudeBinary : nil
    }

    // 3. cc switch db（仅 activeSource == .ccSwitch 时使用）
    if appState.activeSource == .ccSwitch {
        appState.ccSwitchDBURL = try? locateCCSwitchDB()
    }

    // 4. 任一关键路径缺失 → 弹回 Setup
    if appState.claudeBinaryPath == nil {
        showSetupWizard()
        return
    }

    poller.start()
}
```

## 已知陷阱

1. **不要缓存 db 路径超过一次会话。** cc switch 允许用户随时改 override，每次启动 App 都重新解析。
2. **`/bin/zsh -li` 慢。** 首次 Setup 可能要 1-3 秒（加载 nvm/zshrc）。给 UI 加 spinner。
3. **db 文件可能正在被 cc switch 写。** 用 `SQLITE_OPEN_READONLY` + WAL 模式打开。
4. **不要假定 `~/.ccswitch/` 存在。** 老版路径；新版默认 `~/.cc-switch/`；且用户可 override。永远走 `AppPathsResolver`。
5. **activeSource 是动态的。** 用户改了 settings.json 的 env 块会切层级。FSEvents 监听 `settings.json`，变化时重算。
6. **不要假定 Claude.app 在 `/Applications/Claude.app`。** 用 NSWorkspace + bundle id `com.anthropic.clause` 查（实际 bundle id 待确认）。
